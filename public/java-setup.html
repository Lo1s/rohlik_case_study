Windows Installation

Option 1: Using Chocolatey (Recommended) 
# Install Chocolatey (if not installed)
Set-ExecutionPolicy Bypass -Scope Process -Force
[System.Net.ServicePointManager]::SecurityProtocol = [System.Net.ServicePointManager]::SecurityProtocol -bor 3072
iex ((New-Object System.Net.WebClient).DownloadString('https://chocolatey.org/install.ps1'))

# Install Java 17 and Maven
choco install openjdk17 maven -y
Option 2: Manual Installation
  1. Download Java 17 from Adoptium
  2. Download Maven from Apache Maven
  3. Extract both to C:\Program Files\
  4. Set environment variables in System Properties

macOS Installation

Using Homebrew (Recommended) 
# Install Homebrew (if not installed)
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"

# Install Java 17 and Maven
brew install openjdk@17 maven

# Set JAVA_HOME
echo 'export JAVA_HOME=/opt/homebrew/opt/openjdk@17' >> ~/.zshrc
echo 'export PATH="$JAVA_HOME/bin:$PATH"' >> ~/.zshrc
source ~/.zshrc

Linux Installation

Ubuntu/Debian 
# Update package list
sudo apt update

# Install Java 17 and Maven
sudo apt install openjdk-17-jdk maven -y

# Set JAVA_HOME
echo 'export JAVA_HOME=/usr/lib/jvm/java-17-openjdk-amd64' >> ~/.bashrc
source ~/.bashrc
CentOS/RHEL/Fedora 
# Install Java 17 and Maven
sudo dnf install java-17-openjdk-devel maven -y

# Set JAVA_HOME
echo 'export JAVA_HOME=/usr/lib/jvm/java-17-openjdk' >> ~/.bashrc
source ~/.bashrc

üîß Troubleshooting Guide

1

Common Issues

‚ö†Ô∏è Issue: "JAVA_HOME is not defined correctly" 
# Fix JAVA_HOME
# Windows
set JAVA_HOME=C:\Program Files\Java\jdk-17

# macOS/Linux
export JAVA_HOME=/path/to/java-17
export PATH=$JAVA_HOME/bin:$PATH
‚ö†Ô∏è Issue: "Port 8080 already in use" 
# Find and kill process using port 8080
# Windows
netstat -ano | findstr 8080
taskkill /PID [PID_NUMBER] /F

# macOS/Linux
sudo lsof -t -i:8080 | xargs kill -9

# Or change port in application.properties
server.port=8081
‚ö†Ô∏è Issue: "Maven dependencies not found" 
# Clear Maven cache and reinstall
./mvnw clean
rm -rf ~/.m2/repository
./mvnw clean compile
2

Performance Tuning

# Increase JVM memory for better performance
export MAVEN_OPTS="-Xmx2g -Xms1g"

# Run with specific JVM arguments
./mvnw spring-boot:run -Dspring-boot.run.jvmArguments="-Xmx1g -Xms512m"
